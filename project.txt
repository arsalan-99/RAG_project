from langchain_community.document_loaders import TextLoader, PyPDFLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_ollama import OllamaEmbeddings
from qdrant_client import QdrantClient
from qdrant_client.models import VectorParams, Distance, PointStruct
from config import *

def ingest_file(filepath: str) -> int:   
    print(f"\n--- Starting ingest for: {filepath} ---")

    if filepath.endswith(".pdf"):
        docs = PyPDFLoader(filepath).load()
    else:
        docs = TextLoader(filepath).load()
    print(f"âœ“ Loaded {len(docs)} pages/documents")

    chunks = RecursiveCharacterTextSplitter(
        chunk_size=700, chunk_overlap=100
    ).split_documents(docs)
    print(f"âœ“ Created {len(chunks)} chunks")

    embeddings = OllamaEmbeddings(model=EMBEDDING_MODEL)
    print(f"âœ“ Embedding {len(chunks)} chunks...")
    vectors = embeddings.embed_documents([c.page_content for c in chunks])
    print(f"âœ“ Created {len(vectors)} vectors")

    client = QdrantClient(QDRANT_URL)
    existing = [c.name for c in client.get_collections().collections]
    if COLLECTION_NAME not in existing:
        client.create_collection(
            collection_name=COLLECTION_NAME,
            vectors_config=VectorParams(size=len(vectors[0]), distance=Distance.COSINE)
        )
        print(f"âœ“ Created collection: {COLLECTION_NAME}")
    else:
        print(f"âœ“ Collection exists, adding to it")

    current_count = client.count(collection_name=COLLECTION_NAME).count
    client.upsert(
        collection_name=COLLECTION_NAME,
        points=[
            PointStruct(
                id=current_count + i,
                vector=vectors[i],
                payload={"text": chunks[i].page_content, "source": filepath}
            )
            for i in range(len(vectors))
        ]
    )
    print(f"âœ“ Stored {len(vectors)} vectors. Total: {current_count + len(vectors)}")
    print(f"--- Ingest complete ---\n")

    return len(chunks)

from langchain_ollama import OllamaEmbeddings
from qdrant_client import QdrantClient
from openai import OpenAI
from config import *

embeddings = OllamaEmbeddings(model=EMBEDDING_MODEL)
client = QdrantClient(QDRANT_URL)
llm_client = OpenAI(base_url=LLM_BASE_URL, api_key=LLM_API_KEY)

def ask(question):
    query_vector = embeddings.embed_query(question)

    results = client.query_points(
        collection_name=COLLECTION_NAME,
        query=query_vector,
        limit=5
    )

    context = "\n\n".join([r.payload["text"] for r in results.points])

    response = llm_client.chat.completions.create(
        model=LLM_MODEL,
        messages=[
            {
                "role": "user",
                "content": f"""Answer the question first based on the context below, then only if you dont findprovide any additional information you know.
                    Context: {context}
                    Question: {question}"""
            }
        ]
    )

    return response.choices[0].message.content

from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
from fastapi.staticfiles import StaticFiles
from pydantic import BaseModel
from qdrant_client import QdrantClient
from qdrant_client.models import Filter, FieldCondition, MatchValue
from pymongo import MongoClient
from datetime import datetime
from contextlib import asynccontextmanager
from rag import ask
from ingest import ingest_file
from config import *
import shutil, os


# clients
qdrant = QdrantClient(QDRANT_URL)
mongo = MongoClient(MONGO_URL)
db = mongo[MONGO_DB]
files_col = db[MONGO_COLLECTION]


@asynccontextmanager
async def lifespan(app: FastAPI):
    default_files = [
        {"path": "../data/monopoly_rules.txt", "filename": "monopoly_rules.txt"},
        {"path": "../data/syllabus.pdf", "filename": "syllabus.pdf"},
    ]

    for file in default_files:
        existing = files_col.find_one({"filename": file["filename"]})
        if not existing:
            print(f"Ingesting default file: {file['filename']}")
            chunk_count = ingest_file(file["path"])
            files_col.insert_one({
                "filename": file["filename"],
                "uploaded_at": datetime.now().isoformat(),
                "chunks": chunk_count,
                "source_path": file["path"]
            })
            print(f"âœ“ Default file ingested: {file['filename']}")
        else:
            print(f"âœ“ Already exists, skipping: {file['filename']}")

    yield


app = FastAPI(lifespan=lifespan)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)


class Question(BaseModel):
    question: str


@app.post("/upload")
async def upload_file(file: UploadFile = File(...)):
    temp_path = f"temp_{file.filename}"

    with open(temp_path, "wb") as f:
        shutil.copyfileobj(file.file, f)

    chunk_count = ingest_file(temp_path)

    files_col.insert_one({
        "filename": file.filename,
        "uploaded_at": datetime.now().isoformat(),
        "chunks": chunk_count,
        "source_path": temp_path
    })

    os.remove(temp_path)

    return {"message": f"{file.filename} uploaded and stored successfully"}


@app.get("/files")
def get_files():
    files = list(files_col.find({}, {"_id": 0}))
    return {"files": files}


@app.delete("/files/{filename}")
def delete_file(filename: str):
    file_record = files_col.find_one({"filename": filename})
    files_col.delete_one({"filename": filename})

    source_path = file_record["source_path"] if file_record else f"temp_{filename}"

    try:
        qdrant.delete(
            collection_name=COLLECTION_NAME,
            points_selector=Filter(
                must=[
                    FieldCondition(
                        key="source",
                        match=MatchValue(value=source_path)
                    )
                ]
            )
        )
        print(f"âœ“ Deleted vectors for source: {source_path}")
    except Exception as e:
        print(f"Qdrant delete error: {e}")

    return {"message": f"{filename} deleted"}


@app.get("/records")
def get_records():
    try:
        records, _ = qdrant.scroll(
            collection_name=COLLECTION_NAME,
            limit=100,
            with_payload=True,
            with_vectors=False
        )
        return {
            "total": qdrant.count(collection_name=COLLECTION_NAME).count,
            "records": [
                {
                    "id": r.id,
                    "text": r.payload["text"],
                    "source": r.payload.get("source", "unknown")
                }
                for r in records
            ]
        }
    except Exception as e:
        return {"total": 0, "records": [], "error": str(e)}


@app.delete("/records/{record_id}")
def delete_record(record_id: int):
    qdrant.delete(
        collection_name=COLLECTION_NAME,
        points_selector=[record_id]
    )
    return {"message": f"Record {record_id} deleted"}


@app.delete("/reset")
def reset_collection():
    qdrant.delete_collection(collection_name=COLLECTION_NAME)
    files_col.drop()
    return {"message": "All data deleted."}


@app.post("/ask")
def ask_question(body: Question):
    answer = ask(body.question)
    return {"answer": answer}



app.mount("/", StaticFiles(directory="../frontend", html=True), name="frontend")

**GAME OBJECTIVE AND COMPONENTS**
The objective of Monopoly is to become the wealthiest player through buying, renting, and selling real estate, forcing all other players into bankruptcy. The game includes a gameboard, two dice, tokens representing the players, 32 houses, 12 hotels, 16 Chance cards, 16 Community Chest cards, 28 Title Deed cards, play money in various denominations from $1 to $500, and a Banker's tray.
**CHARACTERS AND TOKENS**
Players choose from a variety of tokens to navigate the board. The current standard lineup includes the Race Car, Top Hat, Scottish Terrier, Cat, Penguin, Rubber Ducky, Thimble, and Sack of Money. Throughout the game's history, several classic tokens have been retired, such as the Cannon, Purse, Lantern, Rocking Horse, Iron, Boot, Battleship, Wheelbarrow, Horse & Rider, and T-Rex.
**SETUP**
To begin the game, place the board on a flat surface and shuffle the Chance and Community Chest cards, placing them facedown on their designated board spaces. Each player selects a token and places it on the "GO" space. One player is designated as the Banker, responsible for managing the Bank's money, Title Deeds, Houses, and Hotels, keeping them strictly separate from their personal assets. The Banker distributes exactly $1,500 to each player: two $500s, four $100s, one $50, one $20, two $10s, one $5, and five $1s. Each player then rolls the dice, and the player with the highest total takes the first turn, with play proceeding clockwise.
**HOW TO PLAY**
On a player's turn, they roll both dice and move their token clockwise the exact number of spaces indicated. Upon landing on a space, they must resolve the required action, such as buying property, paying rent, drawing a card, or going to jail. If a player rolls doubles, they resolve the space and then take another turn. However, rolling doubles three consecutive times immediately sends the player to Jail, ending their turn. Once all actions are completed, the dice are passed to the left.
**GAME BOARD SPACES**
The board features several different types of spaces. When landing on unowned property, a player may buy it for the printed price to receive the Title Deed. If they decline, the Banker auctions it to the highest bidder. Landing on owned property requires the player to pay rent based on the Title Deed and the property's development level. Every time a player lands on or passes the "GO" space, they collect $200 from the Bank. Landing on Chance or Community Chest requires drawing the top card, following its instructions, and placing it at the bottom of the deck. Income Tax spaces require a payment of either $200 or 10% of the player's total net worth, while the Luxury Tax demands a flat $75 payment to the Bank. Free Parking is simply a resting space with no action required, and the Just Visiting space is similarly inactive for players passing by. Landing on the "Go To Jail" space moves the token directly to Jail without collecting $200 for passing GO, ending the turn immediately.
**PURCHASING AND RENT RULES**
Rent is significantly affected by the properties a player owns. When a player owns all properties of a single color group, creating a monopoly, the rent on all unimproved properties in that group is doubled. Rent for Railroads increases based on how many the same player owns, ranging from $25 for one to $200 for all four. Utility rent is calculated by multiplying the dice roll by four if one utility is owned, or by ten if both are owned.
**BUILDING RULES**
Players can purchase Houses only when they own a complete color set. Building must be uniform, meaning a second House cannot be placed on any property in a set until one House is on every property in that group. The maximum number of Houses allowed on a single property is four. Once a property has four Houses, the player may upgrade to a Hotel by paying the Hotel cost, returning the four Houses to the Bank, and placing the Hotel. Only one Hotel is permitted per property, and Railroads and Utilities cannot be developed with buildings.
**JAIL RULES**
Players are sent to Jail if they land on "Go To Jail," draw a card explicitly sending them there, or roll doubles three consecutive times in a single turn. While in Jail, players can still buy, sell, trade properties, collect rent, and build. To get out, a player can pay a $50 fine before rolling on their next turn, use a "Get Out of Jail Free" card, or roll doubles on any of their next three turns. If doubles are not rolled by the third turn, the $50 fine must be paid, and the player then moves the number of spaces rolled.
**MORTGAGES AND TRADING**
Unimproved properties, cash, and "Get Out of Jail Free" cards can be traded or sold between players at any time. To raise capital, players can mortgage properties to the Bank for the value printed on the back of the Title Deed. However, all buildings on a color set must be sold back to the Bank at half price before any property in that set can be mortgaged. Mortgaged properties cannot collect rent. To unmortgage, the player pays the Bank the mortgage value plus 10% interest.
**BANKRUPTCY AND WINNING**
A player declares bankruptcy if they owe more money than they can pay to another player or the Bank. Before declaring, they must try to raise funds by selling buildings at half price and mortgaging properties. If still unable to pay a player, all assets are turned over to that creditor, who must immediately pay the 10% interest on inherited mortgaged properties. If the debt is to the Bank, all assets go to the Bank, mortgages are cancelled, and the properties are auctioned to the remaining players. The bankrupt player is eliminated, and the last remaining player is declared the winner.

const API = "http://localhost:8000";

async function loadRecords() {
    const response = await fetch(`${API}/records`);
    const data = await response.json();

    document.getElementById("totalCount").innerText = `Total vectors stored: ${data.total}`;

    const container = document.getElementById("records");
    container.innerHTML = "";

    if (data.records.length === 0) {
        container.innerHTML = "<p>No records found.</p>";
        return;
    }

    data.records.forEach(record => {
        const div = document.createElement("div");
        div.className = "record-card";
        div.innerHTML = `
            <div class="record-meta">
                <span>ID: ${record.id}</span>
                <span>Source: ${record.source}</span>
                <button class="danger small" onclick="deleteRecord(${record.id})">Delete</button>
            </div>
            <p class="record-text">${record.text.substring(0, 200)}...</p>
        `;
        container.appendChild(div);
    });
}

async function deleteRecord(id) {
    if (!confirm(`Delete record ${id}?`)) return;
    await fetch(`${API}/records/${id}`, { method: "DELETE" });
    loadRecords();
}

async function resetDB() {
    if (!confirm("Delete ALL data? This cannot be undone.")) return;
    await fetch(`${API}/reset`, { method: "DELETE" });
    loadRecords();
}

// load on page open
loadRecords();

async function loadFiles() {

    const container = document.getElementById("fileList");
    container.innerHTML = "";

    if (data.files.length === 0) {
        container.innerHTML = "<p>No files uploaded yet.</p>";

    data.files.forEach(file => {
        const div = document.createElement("div");
        div.className = "record-card";
        div.innerHTML = `
            <div class="record-meta">
                <span>ðŸ“„ ${file.filename}</span>

                <span>Uploaded: ${file.uploaded_at ? file.uploaded_at.split("T")[0] : "Unknown"}</span>

        `;
        container.appendChild(div);
    });

async function uploadFile() {
    const fileInput = document.getElementById("fileInput");
    const status = document.getElementById("uploadStatus");

    if (!fileInput.files[0]) {
        status.innerText = "Please select a file first.";
        return;
    }

    const formData = new FormData();
    formData.append("file", fileInput.files[0]);
    status.innerText = "Uploading and processing...";

    const response = await fetch(`${API}/upload`, {
        method: "POST",
        body: formData
    });

    const data = await response.json();
    status.innerText = data.message;
    loadFiles();
}

async function deleteFile(filename) {
    if (!confirm(`Delete ${filename} and all its vectors?`)) return;
    await fetch(`${API}/files/${filename}`, { method: "DELETE" });
    loadFiles();
}

async function uploadFile() {
    const fileInput = document.getElementById("fileInput");
    const status = document.getElementById("uploadStatus");

    if (!fileInput.files[0]) {
        status.innerText = "Please select a file first.";
        return;
    }

    const formData = new FormData();
    formData.append("file", fileInput.files[0]);

    status.innerText = "Uploading and processing...";

    const response = await fetch("http://localhost:8000/upload", {
        method: "POST",
        body: formData
    });

    const data = await response.json();
    status.innerText = data.message;
}

async function askQuestion() {
    const question = document.getElementById("question").value;
    const answerDiv = document.getElementById("answer");

    answerDiv.innerText = "Thinking...";

    const response = await fetch("http://localhost:8000/ask", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ question: question })
    });

    const data = await response.json();
    answerDiv.innerText = data.answer;
}
